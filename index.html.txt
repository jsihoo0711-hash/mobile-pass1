let current=300;

setInterval(()=>{

current--;

if(current<=0){
current=300;
}

document.getElementById("timer_bar").style.width=
(current/total*100)+"%";

document.getElementById("detail_open").onclick=()=>{

document.getElementById("detail_page").style.display="flex";

};

document.getElementById("back_btn").onclick=()=>{

document.getElementById("detail_page").style.display="none";

};

/* 관리자 */

let tap=0;

document.getElementById("top_touch").onclick=()=>{

tap++;

if(tap>=3){

document.getElementById("admin").style.display="flex";

tap=0;

}

setTimeout(()=>tap=0,1000);

};

document.getElementById("apply_btn").onclick=()=>{

const name=
document.getElementById("in_name").value || "김민준";

const birth=
document.getElementById("in_birth").value || "060424-1234567";

const addr=
document.getElementById("in_addr1").value || "서울특별시";

const city=
document.getElementById("in_city").value || "서울시청";

document.getElementById("main_name").innerText=name;

document.getElementById("main_birth").innerText=
birth.split("-")[0];

document.getElementById("main_addr").innerText=addr;

document.getElementById("detail_name").innerText=name;

document.getElementById("detail_birth").innerText=birth;

document.getElementById("detail_addr").innerText=addr;

document.getElementById("city").innerText=city;

document.getElementById("admin").style.display="none";

};

/* 사진 */

document.getElementById("in_photo").onchange=(e)=>{

const file=e.target.files[0];

if(!file) return;

const reader=new FileReader();

reader.onload=(ev)=>{

document.getElementById("main_photo").src=
ev.target.result;

document.getElementById("detail_photo").src=
ev.target.result;

};

reader.readAsDataURL(file);

};
